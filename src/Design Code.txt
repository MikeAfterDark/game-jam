OBJECT Shop {
    num_balls = 4
    Balls {}
    
    Update()
        foreach Ball in Balls{}
            if any are hovered
                freeze their x and y
                open the purchase popup
                purchase button enabled = player ball holder has room and player can afford it
        
        if purchase popup open
            keep track of mouse and close it if purchased or mouse leaves area
    
        if mouse down on handle and not in wheel round
            move horizontally until [#,#] limits 
        // if in wheel round and opened then, | assuming its possible to start a round
        //    close the shop                  | with the shop open lol

        if x pos > #
            opened = true
        else if x pos < # and opened
            Refresh()
            opened = false

    Refresh()
        clears previous Balls{}
        spawns num_balls into Balls{} (set Ball_Location for each)
    
    Ball Hovered(ball)
        throw up the purchase popup
}

OBJECT Ball Holder {
    Max Slots
    is_player_holder
    
    Setup Round()
        gets a list of balls and num unlocked slots

    Next Ball()
        return the next ball in queue, nil if no balls left

    Update()
        if not in wheel round
            foreach Ball in Balls{}
                if any are hovered 
                    open the sell popup
                    purchase button enabled = player ball holder has room and player can afford it
        
        if purchase popup open
            keep track of mouse and close it if sold or mouse leaves area or if start wheel round
    
    Draw()
        renders the unlocked slots
        renders the balls 
        renders the locked slots
}
